    ENDIF

    IF can_save THEN
        {write append code here}
    ELSE
        PRINT "Queen count exceeds daily limit"
    ENDIF

BEGIN
check_queen_count
    INPUT filename

    total_queens ← 0

    IF file does not exist THEN
        RETURN TRUE
    ENDIF

    OPEN filename FOR READING

        READ and discard header row

        FOR each record in the file

            extract the number of queens from the record

            total_queens ← total_queens + number_of_queens

        ENDFOR

    CLOSE filename

    IF total_queens ≥ 10 THEN
        RETURN FALSE
    ELSE
        RETURN TRUE
    ENDIF

END
